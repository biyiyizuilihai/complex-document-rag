**Tech Stack:** Python 3.11+、FastAPI、llama-index、Qdrant、DashScope/OpenAI compatible API、unittest

def _build_image_url_if_exists(image_path: str, artifacts_root: str) -> str:
    """构建图片 URL，文件不存在时返回空字符串以避免前端 404。"""
    if not image_path:
        return ""
    # 先尝试 image_path 是否直接可访问
    if os.path.isfile(image_path):
        return build_artifact_url(image_path, artifacts_root)
    # 再尝试把它拼接到 artifacts_root 下
    candidate = os.path.join(artifacts_root, image_path.lstrip("/"))
    if os.path.isfile(candidate):
        return build_artifact_url(candidate, artifacts_root)
    return ""
```

    def _apply_evidence_fallback(
        self,
        retrieval: dict[str, list[Any]],
        answer_assets: list[Any],
    ) -> list[Any]:
        """当 text_results 和 answer_assets 均为空时，降级使用 filtered retrieval 中最优节点。

        触发条件：text_results=[] 且 answer_assets=[]
        行为：直接取 filtered table_results + image_results 中分数最高的前 2 个（跳过 judge）
        """
        if retrieval.get("text_results") or answer_assets:
            return answer_assets

        candidates = sorted(
            list(retrieval.get("table_results", [])) + list(retrieval.get("image_results", [])),
            key=_safe_score,
            reverse=True,
        )[:2]

        if candidates:
            LOGGER.warning(
                "[fallback] evidence empty — using %d low-confidence node(s) from filtered retrieval",
                len(candidates),
            )
        return candidates
```
